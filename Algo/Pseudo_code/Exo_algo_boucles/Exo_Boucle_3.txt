//Boucle Tant Que //
//VARIABLE//

nombreDepart est entier
i est un entier

Debut Programme

// Demander à l'utilisateur d'entrer un nombre de départ
    Ecrire "Entrer un nombre de depart : "
    Lire nombreDepart
    i <-- 1
Debut Tant Que
    Tant que i <= 10 Faire
    nombreDepart + i
    i <-- i + 1
Fin de Tant Que

Fin Programme

//Boucle Pour//
//VARIABLE//

nombreDepart est entier
i est un entier

// Demander à l'utilisateur d'entrer un nombre de départ
    Ecrire "Entrer un nombre de depart : "
    Lire nombreDepart

Debut Pour
    Pour i de 1 à 10 faire
    nombreDepart + i 
Fin Pour

Fin Programme
